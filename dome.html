import torch
from PIL import Image
import numpy as np

# Load the YOLOv5 model
model = torch.hub.load('ultralytics/yolov5', 'yolov5s', pretrained=True)

def yolo5_predict(image_path):
    # Load the image
    img = Image.open(image_path)
    
    # Perform prediction
    results = model(img)
    
    # Extract the results
    detections = results.xyxy[0]   # detections in (x1, y1, x2, y2, conf, cls)
    
    return detections

# Example usage
# Replace 'path_to_your_image.jpg' with the path to the image you want to predict
detections = yolo5_predict('path_to_your_image.jpg')
print(detections)
